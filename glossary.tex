\usepackage[toc]{glossaries}

\makeglossaries

\newglossaryentry{value function}{
    name=value function,
    description={The expected total reward when acting according to a given policy in a given MDP from a given state}
}

\newglossaryentry{q function}{
    name=q function,
    description={The expected total reward when acting according to a given policy in a given MDP from a given state and action}
}

\newglossaryentry{exploration-exploitation tradeoff}{
    name=exploration-exploitation tradeoff,
    description={The tradeoff between exploiting actions we know to be good and exploring new actions that could be better}
}
